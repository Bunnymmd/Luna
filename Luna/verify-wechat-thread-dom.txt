    document.body.dataset.ready = "boot";

    const openStore = () => new Promise((resolve, reject) => {
      const request = indexedDB.open("LunaDesktopDB");
      request.onupgradeneeded = () => {
        const db = request.result;
        if (!db.objectStoreNames.contains("store")) {
          db.createObjectStore("store");
        }
      };
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error || new Error("open db failed"));
    });

    const putValue = (db, key, value) => new Promise((resolve, reject) => {
      const tx = db.transaction("store", "readwrite");
      tx.objectStore("store").put(value, key);
      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error || new Error(`put ${key} failed`));
    });

    const sleep = (ms) => new Promise((resolve) => setTimeout(resolve, ms));

    const waitFor = async (getter, timeout = 15000) => {
      const start = Date.now();
      while (Date.now() - start < timeout) {
        const value = getter();
        if (value) return value;
        await sleep(120);
      }
      throw new Error("timeout");
    };

    const seedWechatChat = async () => {
      const db = await openStore();
      const now = new Date().toISOString();
      const seededChat = [{
        id: "wechat-chat-verify-xiuxiu-user",
        fromContact: {
          id: "char-verify-xiuxiu",
          type: "CHAR",
          nickname: "秀秀"
        },
        toUser: {
          id: "user-verify",
          type: "USER",
          nickname: "USER"
        },
        message: "刚才看你灯熄了还关了门，不是在等我吗？",
        unread: 0,
        createdAt: now,
        updatedAt: now,
        messages: [
          {
            id: "verify-left-1",
            side: "left",
            type: "text",
            text: "怎么还没睡？",
            createdAt: now,
            updatedAt: now
          },
          {
            id: "verify-left-2",
            side: "left",
            type: "text",
            text: "刚才看你灯熄了还关了门，不是在等我吗？",
            createdAt: now,
            updatedAt: now
          },
          {
            id: "verify-right-1",
            side: "right",
            type: "text",
            text: "我过去干嘛？",
            createdAt: now,
            updatedAt: now
          }
        ]
      }];
      await putValue(db, "isLockEnabled", false);
      await putValue(db, "wechatFriendRequests", []);
      await putValue(db, "wechatChats", seededChat);
      db.close();
    };

    const openThread = async () => {
      document.body.dataset.step = "seed";
      await seedWechatChat();
      document.body.dataset.step = "load-frame";
      const frame = document.getElementById("app-frame");
      frame.src = "/index.html";
      document.body.dataset.step = "wait-frame";
      await waitFor(() => frame.contentDocument?.readyState === "complete" ? frame.contentDocument : null);
      const doc = frame.contentDocument;
      document.body.dataset.step = "wait-dock";
      const dock = await waitFor(() => doc.querySelector(".absolute.bottom-0.left-0.right-0.pointer-events-none.flex.flex-col.items-center.z-50 > div:last-child"));
      const wechatLauncher = dock.children[1];
      document.body.dataset.step = "click-wechat";
      wechatLauncher.dispatchEvent(new MouseEvent("click", { bubbles: true, cancelable: true, view: frame.contentWindow }));
      document.body.dataset.step = "wait-thread";
      const threadCard = await waitFor(() => doc.querySelector(".wechat-thread-card"));
      await sleep(300);
      document.body.dataset.step = "click-thread";
      threadCard.dispatchEvent(new MouseEvent("click", { bubbles: true, cancelable: true, view: frame.contentWindow }));
      await sleep(900);
      document.body.dataset.ready = "true";
      document.body.dataset.step = "done";
    };

    openThread().catch((error) => {
      console.error(error);
      document.body.dataset.ready = "error";
      document.body.dataset.step = error && error.message ? error.message : "unknown";
